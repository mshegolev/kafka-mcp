     - Define a .proto schema:
       ```python

       syntax = "proto3";
       package com.test;
       ```

     - Create `ProtobufSerializer` from `confluent_kafka.schema_registry.protobuf`:
       ```python
       from confluent_kafka.schema_registry.protobuf import ProtobufSerializer
       from confluent_kafka.schema_registry import Schema
       ```
       NOTE: ProtobufSerializer needs a generated protobuf message class, which is
       more complex. **Use the simpler approach:** manually construct Confluent-framed
       Protobuf bytes by:
       a. Register the schema manually via SR client:
          ```python
          schema = Schema(proto_schema_str, schema_type="PROTOBUF")
          schema_id = sr_client.register_schema("test-proto-value", schema)
          ```
       b. Compile the proto and serialize manually using grpc_tools (same approach as
          the adapter's decode path, but in reverse):
          ```python
          import struct, tempfile, os
          from grpc_tools import protoc as _protoc
          from google.protobuf import descriptor_pb2, descriptor_pool, message_factory

          # Compile .proto to FileDescriptorSet
          with tempfile.TemporaryDirectory() as tmpdir:
              proto_path = os.path.join(tmpdir, "schema.proto")
              out_path = os.path.join(tmpdir, "schema.desc")
              _wkt_proto_path = os.path.join(os.path.dirname(_protoc.__file__), "_proto")
              with open(proto_path, "w") as f:
                  f.write(proto_schema_str)
              _protoc.main([
                  "protoc",
                  f"--proto_path={tmpdir}",
                  f"--proto_path={_wkt_proto_path}",
                  f"--descriptor_set_out={out_path}",
                  "--include_imports",
                  proto_path,
              ])
              with open(out_path, "rb") as f:
                  fds = descriptor_pb2.FileDescriptorSet()
                  fds.ParseFromString(f.read())

          pool = descriptor_pool.DescriptorPool()
          file_desc = None
          for file_proto in fds.file:
              file_desc = pool.Add(file_proto)
          msg_desc = list(file_desc.message_types_by_name.values())[0]
          MsgClass = message_factory.GetMessageClass(msg_desc)
          ```
       c. Produce 3 messages with proper Confluent Protobuf framing:
          ```python
          producer = Producer({"bootstrap.servers": bootstrap_servers})
          for i in range(3):
              msg = MsgClass()
              msg.event_id = f"EVT-{i}"
              msg.priority = 10 + i
              msg.description = f"Test event {i}"
              payload = msg.SerializeToString()
              # Confluent Protobuf wire format: magic(0x00) + schema_id(4B BE) + message_index(0x00 for first) + payload
              framed = struct.pack(">bI", 0, schema_id) + b"\x00" + payload
              producer.produce("test-proto", key=f"proto-key-{i}".encode(), value=framed)
          producer.flush(timeout=10)
          time.sleep(2)
          ```
